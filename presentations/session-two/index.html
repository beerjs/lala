const keys = {
    'q': getFreq(0),
    'w': getFreq(2),
    'e': getFreq(3),
    'r': getFreq(5),
    't': getFreq(7),
    'y': getFreq(9),
    'u': getFreq(10),
    'i': getFreq(12),
};
window.addEventListener('keydown', ({key}) => {
   var osc = audioContext.createOscillator();
